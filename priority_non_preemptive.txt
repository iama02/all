PRIORITY SCHEDULING (NON-PREEMPTIVE) – C PROGRAM

Rule:
- Lower priority number indicates higher priority.
- Once a process starts execution, it runs till completion.

C Program:

#include <stdio.h>

int main() {
    int n, t = 0, completed = 0;
    float wt = 0, tat = 0;

    printf("Enter number of processes: ");
    scanf("%d", &n);

    int at[n], bt[n], pr[n], ft[n], ta[n], w[n], done[n];

    for (int i = 0; i < n; i++) {
        printf("Enter AT BT Priority for P%d: ", i + 1);
        scanf("%d %d %d", &at[i], &bt[i], &pr[i]);
        done[i] = 0;
    }

    while (completed < n) {
        int idx = -1;
        int min_pr = 9999;

        for (int i = 0; i < n; i++) {
            if (at[i] <= t && !done[i] && pr[i] < min_pr) {
                min_pr = pr[i];
                idx = i;
            }
        }

        if (idx == -1) {
            t++;
        } else {
            t += bt[idx];
            ft[idx] = t;
            ta[idx] = ft[idx] - at[idx];
            w[idx] = ta[idx] - bt[idx];

            wt += w[idx];
            tat += ta[idx];

            done[idx] = 1;
            completed++;
        }
    }

    printf("\nPID AT BT P FT TAT WT\n");
    for (int i = 0; i < n; i++) {
        printf("P%d  %d  %d  %d  %d  %d  %d\n",
               i + 1, at[i], bt[i], pr[i], ft[i], ta[i], w[i]);
    }

    printf("\nAverage Waiting Time = %.2f", wt / n);
    printf("\nAverage Turnaround Time = %.2f\n", tat / n);

    return 0;
}
