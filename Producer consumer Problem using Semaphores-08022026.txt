#include <stdio.h>
#include <stdlib.h>

// semaphore variables
int mutex = 1;
int full = 0;
int empty = 3;
int item = 0;

// function declarations
void producer();
void consumer();
int wait(int s);
int signal(int s);

int main()
{
    int choice;

    printf("\n1. Producer\n2. Consumer\n3. Exit");

    while(1)
    {
        printf("\nEnter your choice: ");
        scanf("%d", &choice);

        if(choice == 1)
        {
            if(mutex == 1 && empty != 0)
                producer();
            else
                printf("Buffer is full!!\n");
        }
        else if(choice == 2)
        {
            if(mutex == 1 && full != 0)
                consumer();
            else
                printf("Buffer is empty!!\n");
        }
        else if(choice == 3)
        {
            exit(0);
        }
        else
        {
            printf("Invalid choice!\n");
        }
    }

    return 0;
}

// wait function
int wait(int s)
{
    s = s - 1;
    return s;
}

// signal function
int signal(int s)
{
    s = s + 1;
    return s;
}

// producer
void producer()
{
    mutex = wait(mutex);
    full  = signal(full);
    empty = wait(empty);

    item = item + 1;
    printf("Producer produces item %d\n", item);

    mutex = signal(mutex);
}

// consumer
void consumer()
{
    mutex = wait(mutex);
    full  = wait(full);
    empty = signal(empty);

    printf("Consumer consumes item %d\n", item);
    item = item - 1;

    mutex = signal(mutex);
}
