#include <stdio.h>

int main()
{
    int n, i, time = 0, tq;
    int at[100], bt[100], rt[100], ft[100];
    int completed = 0;
    float tat = 0, wt = 0;

    printf("Enter number of processes: ");
    scanf("%d", &n);

    for(i=0;i<n;i++)
    {
        printf("\nEnter Arrival Time and Burst Time for P%d: ", i+1);
        scanf("%d %d",&at[i],&bt[i]);
        rt[i] = bt[i];
    }

    printf("\nEnter Time Quantum: ");
    scanf("%d",&tq);

    printf("\nProcess\tAT\tBT\tFT\tTAT\tWT\n");

    while(completed < n)
    {
        int found = 0;

        for(i=0;i<n;i++)
        {
            if(at[i] <= time && rt[i] > 0)
            {
                found = 1;

                if(rt[i] > tq)
                {
                    time += tq;
                    rt[i] -= tq;
                }
                else
                {
                    time += rt[i];
                    rt[i] = 0;
                    completed++;

                    ft[i] = time;
                    int TAT = ft[i] - at[i];
                    int WT = TAT - bt[i];

                    tat += TAT;
                    wt += WT;

                    printf("P%d\t%d\t%d\t%d\t%d\t%d\n",
                           i+1, at[i], bt[i], ft[i], TAT, WT);
                }
            }
        }

        if(found == 0)
            time++;   // CPU Idle
    }

    printf("\nAverage Waiting Time = %.2f", wt/n);
    printf("\nAverage Turnaround Time = %.2f\n", tat/n);

    return 0;
}
